. dns server
#yum install bind
#nano /etc/named.conf
options {
    listen-on port 53 { 172.16.1.10; };
    directory "/var/named";
    allow-query { localhost; 172.16.1.0/24 }
};

zone "example.com" {
    type master;
    file "/var/named/db.osc";
    allow-query { any; }
};

zone "1.16.172.in-addr.arpa" {
    type master;
    file "/var/named/db.172";
    allow-query { any; }
};


#nano /var/named/db.osc
$TTL 86400      ; 1 day
@ IN SOA ns1.example.com. admin.example.com. (
    2023092201  ; serial number
    3600        ; refresh (1 hour)
    1800        ; retry (30 minutes)
    604800      ; expire (1 week)
    86400       ; minimum (1 day)
)

; Name servers
@ IN NS ns1.example.com.
@ IN NS ns2.example.com.

; DNS records
ns1 IN A 192.168.0.1
ns2 IN A 192.168.0.2
www IN A 192.168.0.3
mail IN A 192.168.0.4

#nano /var/named/db.172
$TTL 86400
@       IN      SOA     ns1.example.com. admin.example.com. (
                        2023092301 ; Serial
                        86400      ; Refresh
                        7200       ; Retry
                        3600000    ; Expire
                        86400 )    ; Minimum TTL

        IN      NS      ns1.example.com.

10       IN      PTR     host1.example.com.
11       IN      PTR     host2.example.com.

#systemctl restart named
